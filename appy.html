import streamlit as st import requests import base64 # --- NEW: Custom Colors using CSS --- st.markdown(""" """, unsafe_allow_html=True) st.title("🎓 Next-Gen AI Student Toolkit") st.sidebar.title("AI Features") selected_tool = st.sidebar.radio( "Choose a Tool:", ["General Chat", "ATS Resume Builder", "Career Path Suggester", "Interview Prep", "📂 View Saved Resumes"] ) # --------------------------------------------------------- # FEATURE: VIEW SAVED RESUMES # --------------------------------------------------------- if selected_tool == "📂 View Saved Resumes": st.markdown("### 📂 Your Saved Resumes") try: # Ask the backend for all saved resumes res = requests.get("http://127.0.0.1:8000/get_resumes") if res.status_code != 200: st.error(f"Error fetching resumes: {res.status_code}") saved_resumes = [] else: saved_resumes = res.json() if not saved_resumes: st.info("No resumes saved yet. Go to the ATS Resume Builder to create one!") else: for resume in saved_resumes: resume_name = resume.get('name', 'Unknown') with st.expander(f"Resume: {resume_name}"): # Show the photo if it exists photo_data = resume.get('photo') if photo_data: try: # Decode the text back into an image image_bytes = base64.b64decode(photo_data) st.image(image_bytes, width=150) except Exception: st.warning("Failed to display resume photo.") # Show the text st.markdown(resume.get('text', 'No resume text available.')) except Exception as e: st.error("Could not load resumes. Is the backend running?") # --------------------------------------------------------- # FEATURE: ATS RESUME BUILDER (With Photo Upload & Save) # --------------------------------------------------------- elif selected_tool == "ATS Resume Builder": st.markdown("### 📄 ATS Resume Builder") # New: Photo Uploader uploaded_photo = st.file_uploader("Upload a Professional Photo (Optional)", type=["png", "jpg", "jpeg"]) user_input = st.text_area("List your education and skills:", height=150) if st.button("Generate AI Resume"): if user_input.strip(): with st.spinner("Writing your resume..."): res = requests.post( "http://127.0.0.1:8000/generate", json={"tool": "resume", "text": user_input} ) if res.status_code == 200: # Save the generated text into Streamlit's temporary memory st.session_state['generated_resume'] = res.json().get("response", "No response content.") else: st.error(f"Failed to generate resume: {res.status_code}") st.session_state['generated_resume'] = f"API Error: {res.text}" # Convert the uploaded photo into base64 text so we can send it easily if uploaded_photo is not None: photo_bytes = uploaded_photo.getvalue() base64_photo = base64.b64encode(photo_bytes).decode('utf-8') st.session_state['photo_data'] = base64_photo else: st.session_state['photo_data'] = "" # If a resume has been generated, show it and offer to save it if 'generated_resume' in st.session_state: st.markdown("---") st.markdown("### Your Generated Resume") # Display the photo if uploaded photo_data = st.session_state.get('photo_data', '') if photo_data: try: image_bytes = base64.b64decode(photo_data) st.image(image_bytes, width=150) except Exception: st.warning("Failed to display uploaded photo.") st.markdown(st.session_state.get('generated_resume', '')) st.markdown("---") # Save Section student_name = st.text_input("Enter your name to save this resume:") if st.button("Save to Database"): if student_name: save_data = { "student_name": student_name, "resume_text": st.session_state.get('generated_resume', ''), "photo_data": photo_data } save_res = requests.post("http://127.0.0.1:8000/save_resume", json=save_data) if save_res.status_code == 200: st.success(save_res.json().get("message", "Resume saved successfully!")) else: st.error(f"Failed to save resume: {save_res.status_code}") else: st.warning("Please enter your name before saving.") # --------------------------------------------------------- # OTHER FEATURES (Chat, Career, Interview) # --------------------------------------------------------- else: tool_mapping = { "General Chat": "chat", "Career Path Suggester": "career", "Interview Prep": "interview" } if selected_tool == "General Chat": st.markdown("### 💬 Developer Chatbot") elif selected_tool == "Career Path Suggester": st.markdown("### 🚀 Career Path Suggester") else: st.markdown("### 🎤 Interview Prep") user_input = st.text_area("Input Details:", height=150) if st.button("Generate AI Response"): if user_input.strip(): with st.spinner("Processing..."): res = requests.post( "http://127.0.0.1:8000/generate", json={"tool": tool_mapping[selected_tool], "text": user_input} ) if res.status_code == 200: st.markdown(res.json().get("response", "No response returned.")) else: st.error(f"Error {res.status_code}: API call failed.")
